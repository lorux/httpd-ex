--pa con stazioni e int
select pa.ID_DOMINIO,staz.ID_STAZIONE,staz.PASSWORD,intpa.ID_INTERMEDIARIO_PA from FESPPROD60.NODO4_CFG.PA pa 
join FESPPROD60.NODO4_CFG.PA_STAZIONE_PA paspa on paspa.FK_PA = pa.OBJ_ID
join FESPPROD60.NODO4_CFG.STAZIONI staz on paspa.FK_STAZIONE = staz.OBJ_ID
join FESPPROD60.NODO4_CFG.INTERMEDIARI_PA intpa on staz.FK_INTERMEDIARIO_PA = intpa.OBJ_ID

--psp con canali e int
select psp.ID_PSP,can.ID_CANALE,can.PASSWORD,intpsp.ID_INTERMEDIARIO_PSP from FESPPROD60.NODO4_CFG.PSP psp 
join FESPPROD60.NODO4_CFG.PSP_CANALE_TIPO_VERSAMENTO_CANALE pspcan on pspcan.FK_PSP = psp.OBJ_ID
join FESPPROD60.NODO4_CFG.CANALI can on pspcan.FK_CANALE = can.OBJ_ID
join FESPPROD60.NODO4_CFG.INTERMEDIARI_PSP intpsp on can.FK_INTERMEDIARIO_PSP = intpsp.OBJ_ID
